CREATE TRIGGER updateTimestamp
    AFTER UPDATE OF pname, goals ON players
    REFERENCING OLD AS OLDROW
    FOR EACH ROW
        UPDATE players 
            SET TIMESTAMP = CURRENT_TIMESTAMP
            WHERE ID = OLDROW.ID;
